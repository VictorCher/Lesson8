using System;

namespace ConsoleApplication1
{
    class Program
    {
        static int Fact(int n)
        {
            int res = 1;
            for (int i = 1; i <= n; i++)
            {
                res = res * i;
            }
            return res;
        }
        static void Main(string[] args)
        {
            int n = int.Parse(Console.ReadLine());
            if (n == 0) Console.WriteLine("Результат: 1");
            else Console.WriteLine("Результат:" + Fact(n));
            Console.ReadKey();
        }
    }
}
